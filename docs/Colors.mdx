import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks'
import theme from './dist/theme.config.json'

export const ColorsList = () => {
  if (!theme) {
    return null;
  }

  const exclude = ['current', 'currentColor', 'transparent']
  const colors = Object.entries(theme.colors)
    .filter(([k]) => !exclude.includes(k))
    .map(([key, value]) => {
      const colorSet = {}
      if (typeof value === 'string') {
        colorSet[key] = value
      } else {
        Object.entries(value).forEach(([k, v]) => {
          if (parseInt(k)) {
            colorSet[k] = v
          } else if (k === 'DEFAULT') {
            colorSet[key] = v
          } else {
            colorSet[`${key}-${k}`] = v
          }
        })
      }

      return {
        title: `${key.charAt(0).toUpperCase()}${key.slice(1)}`,
        subtitle: `colors.${key}`,
        colors: colorSet
      }
    })

  return colors?.map((color) => (<ColorItem key={color.title} title={color.title} subtitle={color.subtitle} colors={color.colors} />))
}

<Meta title="Foundations/Colors" />

# Colors

<ColorPalette>
  <ColorsList />
</ColorPalette>
